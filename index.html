function extrairDadosExcel(data, fileName) { let workbook = XLSX.read(data, { type: 'array' }); let primeiraPlanilha = workbook.Sheets[workbook.SheetNames[0]]; let linhas = XLSX.utils.sheet_to_json(primeiraPlanilha, { header: 1, defval: "" }); let nfe = 'NFE_' + Date.now(); let itens = []; let eansEncontrados = new Map(); // Procura a NFe de forma simples for (let i = 0; i < Math.min(20, linhas.length); i++) { let linha = linhas[i]; if (linha && linha.length) { for (let j = 0; j < linha.length; j++) { let valor = linha[j]; if (valor && typeof valor === 'string') { let nums = valor.match(/\d{6,9}/); if (nums && nums[0] && nums[0].length >= 6) { nfe = nums[0]; break; } } } } if (nfe !== 'NFE_' + Date.now()) break; } // Extrai os produtos for (let i = 2; i < linhas.length; i++) { let linha = linhas[i]; if (!linha || linha.length < 8) continue; let codigo = linha[0] ? String(linha[0]).trim() : ''; let descricao = linha[1] ? String(linha[1]).trim() : ''; let eanRaw = linha[2] ? String(linha[2]).trim() : ''; // Extrai EAN let ean = ''; let eanMatch = eanRaw.match(/\*(\d+)\*/); if (eanMatch) ean = eanMatch[1]; if (!ean) ean = eanRaw.replace(/\D/g, ''); if (ean.length > 13) ean = ean.substring(0, 13); let qtd = 0; if (linha[7] !== undefined && linha[7] !== null) { let qtdStr = String(linha[7]).replace('.', '').replace(',', '.').trim(); qtd = parseFloat(qtdStr) || 0; } let custo = 0; if (linha[11] !== undefined && linha[11] !== null) { let custoStr = String(linha[11]).replace('.', '').replace(',', '.').trim(); custo = parseFloat(custoStr) || 0; } if (ean && ean.length >= 10 && qtd > 0 && !eansEncontrados.has(ean)) { eansEncontrados.set(ean, { ean: ean, codigo: codigo, desc: descricao.substring(0, 50) || 'Sem descrição', qtd: qtd, custo: custo, conf: 0 }); } } return { nfe: nfe, itens: Array.from(eansEncontrados.values()), fileName: fileName }; }
